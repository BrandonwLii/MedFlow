A* Pathfinding Algorithm
function findPath(start, goal, grid) {
  const openSet = new PriorityQueue();
  const gScore = new Map();  // cost from start
  const fScore = new Map();  // g + heuristic

  gScore.set(start, 0);
  fScore.set(start, heuristic(start, goal));

  while (!openSet.isEmpty()) {
    const current = openSet.pop();
    if (current === goal) return path;

    for (const neighbor of getNeighbors(current)) {
      const tentativeG = gScore.get(current) + 1;
      if (tentativeG < gScore.get(neighbor)) {
        // Found a better path
        cameFrom.set(neighbor, current);
        gScore.set(neighbor, tentativeG);
        fScore.set(neighbor, tentativeG + heuristic(neighbor, goal));
      }
    }
  }
}

Smart Job Dispatch
function assignBestAgent(job, agents) {
  // Filter eligible agents
  const eligible = agents.filter(agent =>
    agent.status === 'IDLE' &&
    agent.battery > 20 &&
    hasAccess(agent, job.dropoff.roomId) &&
    matchesPool(agent.pool, job.priority)
  );

  // Score each agent
  return eligible.reduce((best, agent) => {
    const distance = manhattanDistance(
      agent.position,
      job.pickup.position
    );
    const score = distance +
      (100 - agent.battery) * 0.5;

    return score < best.score
      ? { agent, score }
      : best;
  }, { agent: null, score: Infinity });
}
